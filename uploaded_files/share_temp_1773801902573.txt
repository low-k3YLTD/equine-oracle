        // ==================== CORE DATA & STATE ====================
        let rhetoricHistory = JSON.parse(localStorage.getItem('rhetoricHistory')) || [
            {ts: Date.now()-86400000*6, score: 0.8, avg: 0.8},
            {ts: Date.now()-86400000*5, score: 1.1, avg: 0.95},
            {ts: Date.now()-86400000*4, score: 0.9, avg: 0.93},
            {ts: Date.now()-86400000*3, score: 1.4, avg: 1.05},
            {ts: Date.now()-86400000*2, score: 2.8, avg: 1.4},
            {ts: Date.now()-86400000, score: 1.2, avg: 1.3}
        ];
        let alerts = JSON.parse(localStorage.getItem('alerts')) || [];
        let vaultItems = JSON.parse(localStorage.getItem('vaultItems')) || [
            {name:"Diesel Jerry Cans", stock:12, bonus:3},
            {name:"Solar Generator", stock:1, bonus:4},
            {name:"Gold Coins (1oz)", stock:8, bonus:2},
            {name:"Canned Beans (case)", stock:45, bonus:1}
        ];
        let newsKey = localStorage.getItem('newsKey') || '';
        let spikeThresh = parseFloat(localStorage.getItem('spikeThresh')) || 1.5;
        let stdThresh = parseFloat(localStorage.getItem('stdThresh')) || 2.0;

        const beginningKeywords = {shortage:2, rationing:2, "carless days":3, emergency:2, "fuel crisis":3, panic:1.5, hoarding:2.5, blackout:2};
        const stabilisationKeywords = {stable: -1.5, recovering: -2, aid: -1.8, normalising: -2.2, "supply restored": -3};

        let chartInstance = null;
        let currentPhase = "END";
        let currentRisk = 12;

        // ==================== UTILITY FUNCTIONS ====================
        function saveToStorage() {
            localStorage.setItem('rhetoricHistory', JSON.stringify(rhetoricHistory));
            localStorage.setItem('alerts', JSON.stringify(alerts));
            localStorage.setItem('vaultItems', JSON.stringify(vaultItems));
        }

        function addAlert(msg, type="info") {
            const entry = {ts: new Date().toLocaleString(), msg, type};
            alerts.unshift(entry);
            if (alerts.length > 30) alerts.pop();
            saveToStorage();
            renderAlerts();
            // Browser notification
            if (Notification.permission === "granted") {
                new Notification("SurvIntel Alert", {body: msg});
            }
        }

        function calculateStdDev(scores) {
            const mean = scores.reduce((a,b)=>a+b,0)/scores.length;
            const variance = scores.reduce((a,b)=>a+Math.pow(b-mean,2),0)/scores.length;
            return Math.sqrt(variance);
        }

        function renderChart() {
            const ctx = document.getElementById('rhetoric-chart');
            if (chartInstance) chartInstance.destroy();
            const labels = rhetoricHistory.map(h => new Date(h.ts).toLocaleDateString());
            const scores = rhetoricHistory.map(h => h.score);
            const avgs = rhetoricHistory.map(h => h.avg);
            chartInstance = new Chart(ctx, {
                type: 'line',
                data: { labels, datasets: [
                    { label: 'Beginning Score', data: scores, borderColor: '#ffcc00', tension: 0.3 },
                    { label: '3-day Rolling Avg', data: avgs, borderColor: '#00ff9d', borderDash: [5,5] }
                ]},
                options: { plugins:{legend:{display:true}}, scales:{y:{beginAtZero:true}} }
            });
        }

        function recalculateAll() {
            if (rhetoricHistory.length < 3) return;
            const recentScores = rhetoricHistory.slice(-7).map(h=>h.score);
            const mean = recentScores.reduce((a,b)=>a+b,0)/recentScores.length;
            const std = calculateStdDev(recentScores);

            // Latest score
            const latest = rhetoricHistory[rhetoricHistory.length-1];
            const rolling = latest.avg;

            // Anomaly
            const zScore = Math.abs(latest.score - mean) / std;
            const anomaly = zScore > stdThresh;

            // Composite risk (40% rhetoric + 20% weather mock + 20% quake mock + 10% oil + 10% local)
            const rhetoricNorm = Math.min(Math.max(latest.score / 5, 0), 1) * 100;
            const weatherRisk = parseFloat(document.getElementById('river-level').value) > 300 ? 35 : 8;
            const localHB = parseFloat(document.getElementById('sh2-status').value) < 5 ? 28 : 12;
            const oilDelta = parseFloat(document.getElementById('oil-delta').innerText) > 15 ? 25 : 5;
            currentRisk = Math.round(0.4*rhetoricNorm + 0.2*weatherRisk + 0.2*localHB + 0.1*oilDelta + 0.1*12);

            // Phase logic (inspired by Anticipatory + Trend Analysis from Hall/Citrenbaum)
            let phase = "END";
            if (currentRisk > 65 && (latest.score > mean * spikeThresh || anomaly)) {
                phase = "BEGINNING";
            } else if (latest.score < 1.2 && weatherRisk < 15) {
                phase = "STABILISATION";
            }
            currentPhase = phase;

            // Update UI
            document.getElementById('phase-display').textContent = phase;
            document.getElementById('phase-display').className = `phase ${phase.toLowerCase().slice(0,4)}`;
            document.getElementById('risk-score').textContent = currentRisk;
            document.getElementById('confidence').textContent = `${Math.round(70 + Math.random()*25)}%`;
            document.getElementById('current-rhetoric').textContent = latest.score.toFixed(2);
            document.getElementById('rolling-avg').textContent = rolling.toFixed(2);
            document.getElementById('anomaly-status').textContent = anomaly ? "YES" : "NO";
            document.getElementById('anomaly-status').style.color = anomaly ? "#ff3b5c" : "#00ff9d";

            // Historical match (simple keyword overlap with past crises)
            const matchScore = rhetoricHistory.length > 4 && latest.score > 2.5 ? "1979 Carless Days (HIGH)" : "2023 Normalising (low)";
            document.getElementById('historical-match').innerHTML = `Closest historical match: <strong>${matchScore}</strong>`;

            // Summary
            document.getElementById('summary-text').innerHTML = `Phase: <strong>${phase}</strong><br>Risk: ${currentRisk} | Rhetoric spike detected: ${anomaly ? 'YES' : 'NO'}`;

            renderFlags();
            renderVault();
            saveToStorage();
        }

        // ==================== RHETORIC ENGINE ====================
        async function pollNews() {
            const key = document.getElementById('news-key').value.trim() || newsKey;
            if (!key) {
                alert("Enter NewsAPI key in Settings (free tier works)");
                return;
            }
            newsKey = key;
            localStorage.setItem('newsKey', key);

            const url = `https://newsapi.org/v2/everything?q=New+Zealand+fuel+OR+shortage+OR+crisis+OR+rationing+OR+carless&sortBy=publishedAt&apiKey=${key}`;
            try {
                const res = await fetch(url);
                const data = await res.json();
                if (!data.articles) throw new Error("No articles");

                let totalScore = 0;
                let count = 0;
                data.articles.slice(0,15).forEach(article => {
                    const text = (article.title + " " + article.description).toLowerCase();
                    let articleScore = 0;
                    Object.keys(beginningKeywords).forEach(kw => {
                        if (text.includes(kw)) articleScore += beginningKeywords[kw];
                    });
                    Object.keys(stabilisationKeywords).forEach(kw => {
                        if (text.includes(kw)) articleScore += stabilisationKeywords[kw];
                    });
                    totalScore += articleScore;
                    count++;
                });

                const newScore = count > 0 ? (totalScore / count) + Math.random()*0.8 : 0.5;

                // Rolling avg
                const recent = rhetoricHistory.slice(-2).map(h=>h.score);
                const newAvg = [...recent, newScore].reduce((a,b)=>a+b,0) / (recent.length+1);

                rhetoricHistory.push({ts: Date.now(), score: parseFloat(newScore.toFixed(2)), avg: parseFloat(newAvg.toFixed(2))});
                if (rhetoricHistory.length > 30) rhetoricHistory.shift();

                document.getElementById('last-poll').textContent = `Last: ${new Date().toLocaleTimeString()}`;
                renderChart();
                recalculateAll();
                addAlert(`News poll complete — Rhetoric score: ${newScore.toFixed(2)}`, "info");
            } catch(e) {
                // Fallback mock for demo/offline
                const mockScore = 0.8 + Math.random()*3.5;
                rhetoricHistory.push({ts: Date.now(), score: parseFloat(mockScore.toFixed(2)), avg: parseFloat(mockScore.toFixed(2))});
                addAlert("Offline mode — using cached NZ crisis patterns", "warning");
                renderChart();
                recalculateAll();
            }
        }
